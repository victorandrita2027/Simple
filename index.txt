<script>
  document.getElementById('sheetForm').addEventListener('submit', function(e) {
    e.preventDefault(); // Stop the form from performing a traditional page reload

    // --- PASTE YOUR EXACT GOOGLE APPS SCRIPT WEB APP URL HERE ---
    const scriptUrl = 'https://script.google.com/macros/s/AKfycbycbI3RxpxhnRslrJYyBRdknsF9emotA6VoYBRql7h2b6G6qXBeyrhV3yY06hM8KUQtFg/exec';

    const submitButton = document.getElementById('submitBtn');
    const statusMessage = document.getElementById('statusMessage');

    // UI Updates during submission
    submitButton.disabled = true;
    submitButton.innerText = 'Sending...';
    statusMessage.style.color = '#333';
    statusMessage.innerText = 'Submitting your data...';

    // Pack the data to match your Google Apps Script keys
    const formData = {
      name: document.getElementById('username').value,
      email: document.getElementById('useremail').value
    };

    // Make the direct network request
    fetch(scriptUrl, {
      method: 'POST',
      mode: 'no-cors', // Crucial to bypass Google's CORS redirect constraints
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(formData)
    })
    .then(() => {
      // Due to 'no-cors', we safely assume success if the network request resolves
      statusMessage.style.color = 'green';
      statusMessage.innerText = 'Success! Data added to Google Sheets.';
      document.getElementById('sheetForm').reset();
    })
    .catch(error => {
      console.error('Submission error:', error);
      statusMessage.style.color = 'red';
      statusMessage.innerText = 'Error sending data. Please check connection.';
    })
    .finally(() => {
      // Reset button state
      submitButton.disabled = false;
      submitButton.innerText = 'Submit Data';
    });
  });
</script>
